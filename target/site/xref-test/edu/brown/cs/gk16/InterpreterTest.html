18      assert (tokens.size() == 4);

45    @Test
46    public void testParseNeighborCommandNormalFunctionality() {
47      // Creating a new stars object to pass into the function
48      StarsObject stars = new StarsObject(3);
49  
50      List<String> commands = new ArrayList<String>();
51      commands.add("neighbors");
52      commands.add("3");
53      commands.add("\"Sol\"");
54  
55      stars.addStars("data/stars/ten-star.csv");
56  
57      List<String> results =
58          Interpreter.parseNeighborsCommand(true, commands, stars);
59      assert (results.size() == 3);
60      assert (results.get(0).equals("70667"));
61      assert (results.get(1).equals("71454"));
62      assert (results.get(2).equals("71457"));
63  
64      commands.remove(2);
65      commands.add("1");
66      commands.add("1");
67      commands.add("1");
68  
69      results = Interpreter.parseNeighborsCommand(true, commands, stars);
70      assert (results.size() == 3);
71      assert (results.get(0).equals("0"));
72      assert (results.get(1).equals("70667"));
73      assert (results.get(2).equals("71454"));

75    }

323   @Test
324   public void testParseToggleCommand() {
325     // Testing normal functionality for parseToggleCommand function
326     AcObject ac = new AcObject();
327     List<String> prefix = new ArrayList<String>();
328     prefix.add("prefix");
329     List<String> whitespace = new ArrayList<String>();
330     whitespace.add("whitespace");
331     List<String> smart = new ArrayList<String>();
332     smart.add("smart");
333     assert (Interpreter.parseToggleCommand(prefix, ac).equals("prefix off"));
334     assert (Interpreter.parseToggleCommand(whitespace, ac)
335         .equals("whitespace off"));
336     assert (Interpreter.parseToggleCommand(smart, ac).equals("smart off"));
337     prefix.add("on");
338     whitespace.add("on");
339     smart.add("on");
340     Interpreter.parseToggleCommand(prefix, ac);
341     Interpreter.parseToggleCommand(whitespace, ac);
342     Interpreter.parseToggleCommand(smart, ac);
343     assert (ac.getPrefix().equals("prefix on"));
344     assert (ac.getWhiteSpace().equals("whitespace on"));
345     assert (ac.getSmart().equals("smart on"));
346   }
