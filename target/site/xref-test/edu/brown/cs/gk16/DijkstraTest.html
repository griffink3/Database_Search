16    @Test
17    public void testFindShortestPath() {
18      // Creating new bacon object and loading database
19      BaconObject bacon = new BaconObject();
20      bacon.addDb("data/bacon/smallbacon.sqlite3");
21      EdgeSetGraph<String> graph = bacon.getGraph();
22      String actor1 = "Will Smith";
23      String actor2 = "Stephen Root";
24      List<String> output = bacon.connect(actor1, actor2);
25      assert (output.size() == 1);
26      assert (output.get(0).equals("Will Smith -> Stephen Root : Jersey Girl"));
27    }

29    @Test
30    public void testFindShortestPathLonger() {
31      // Creating new bacon object and loading database
32      BaconObject bacon = new BaconObject();
33      bacon.addDb("data/bacon/smallbacon.sqlite3");
34      String actor1 = "Arnold Schwarzenegger";
35      String actor2 = "Willow Smith";
36      List<String> output = bacon.connect(actor1, actor2);
37      assert (output.get(0).equals(
38          "Arnold Schwarzenegger -> Steve Coogan : Around the World in 80 Days"));
39      assert (output.get(1).equals("Steve Coogan -> Cate Blanchett : Hot Fuzz"));
40      assert (output.get(2).equals("Cate Blanchett -> Betty White : Ponyo"));
41      assert (output.get(3).equals("Betty White -> Willow Smith : The Lorax"));
42    }
