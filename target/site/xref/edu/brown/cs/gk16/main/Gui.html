72    /**
73     * Handle requests to search for nearest neighbors.
74     */
75    public class NeighborsHandler implements TemplateViewRoute {
76      @Override
77      public ModelAndView handle(Request req, Response res) {
78        String message = "";
79        if (!starsLoaded) {
80          message = "Error occurred: no stars have been loaded.";
81        } else {
82          QueryParamsMap qm = req.queryMap();
83          String numNeighbors = qm.value("numNeighbors");
84          String searchParam = qm.value("searchParam");
85          List<String> commands = new ArrayList<String>();
86          commands.add("neighbors");
87          commands.add(numNeighbors);
88          Matcher m =
89              Pattern.compile("([^\"]\\S*|\".+?\")\\s*").matcher(searchParam);
90          while (m.find()) {
91            commands.add(m.group(1));
92          }
93          List<String> nearestNeighbors =
94              Interpreter.parseNeighborsCommand(starsLoaded, commands, stars);
95          if (nearestNeighbors.isEmpty()) {
96            message = "Error occurred: fix input (or no stars found)";
97          } else {
98            message = "The " + numNeighbors + " closest neighbors to "
99                + searchParam + " are ";
100           for (String neighbor : nearestNeighbors) {
101             if (!stars.getStarById(neighbor).getName().equals("")) {
102               neighbor = stars.getStarById(neighbor).getName();
103             }
104             message = message + neighbor + ", ";
105           }
106         }
107       }
108       Map<String, Object> variables = ImmutableMap.of("title",
109           "Stars: Query the database", "message", message);
110       return new ModelAndView(variables, "query.ftl");
111     }
112   }

191   /**
192    * Handles request to update autocorrect settings.
193    */
194   public class ToggleHandler implements TemplateViewRoute {
195     @Override
196     public ModelAndView handle(Request req, Response res) {
197       QueryParamsMap qm = req.queryMap();
198       String prefix = qm.value("Prefix");
199       String whitespace = qm.value("WhiteSpace");
200       String smart = qm.value("Smart");
201       String led = qm.value("led");
202       String ngram = qm.value("nGram");
203       String message = "";
204       List<String> prefComm = new ArrayList<String>();
205       prefComm.add("prefix");
206       prefComm.add(prefix);
207       Interpreter.parseToggleCommand(prefComm, ac);
208       List<String> wsComm = new ArrayList<String>();
209       wsComm.add("whitespace");
210       wsComm.add(whitespace);
211       Interpreter.parseToggleCommand(wsComm, ac);
212       List<String> smartComm = new ArrayList<String>();
213       smartComm.add("smart");
214       smartComm.add(smart);
215       Interpreter.parseToggleCommand(smartComm, ac);
216       List<String> ledComm = new ArrayList<String>();
217       ledComm.add("led");
218       ledComm.add(led);
219       int result1;
220       if (led.equals("")) {
221         result1 = 1;
222       } else {
223         result1 = Interpreter.parseLedAndNCommand(ledComm, ac);
224       }
225       List<String> nComm = new ArrayList<String>();
226       nComm.add("n");
227       nComm.add(ngram);
228       int result2;
229       if (ngram.equals("")) {
230         result2 = 1;
231       } else {
232         result2 = Interpreter.parseLedAndNCommand(nComm, ac);
233       }
234       if (result1 < 0 || result2 < 0) {
235         message = "Error occurred: make sure input is a positive integer";
236       } else {
237         message = "Prefix " + prefix + ", Whitespace " + whitespace + ", Smart "
238             + smart + ", Led " + led + ", N-Gram " + ngram;
239       }
240       Map<String, Object> variables = ImmutableMap.of("title",
241           "Autocorrect: Text Prediction", "message", message);
242       return new ModelAndView(variables, "ac_input.ftl");
243     }

369   /**
370    * Handles request to get an actor page.
371    */
372   public class GetActorHandler implements TemplateViewRoute {
373     @Override
374     public ModelAndView handle(Request req, Response res) {
375       QueryParamsMap qm = req.queryMap();
376       String name = req.params(":name");
377       StringBuffer s = new StringBuffer(name);
378       for (int i = 0; i < s.length(); i++) {
379         if (s.charAt(i) == '-') {
380           s.setCharAt(i, '\'');
381         } else if (s.charAt(i) == '_') {
382           s.setCharAt(i, ' ');
383         } else if (s.charAt(i) == '=') {
384           s.setCharAt(i, '\"');
385         } else if (s.charAt(i) == '+') {
386           s.setCharAt(i, '.');
387         }
388       }
389       name = s.toString();
390       films = bacon.queryTableList(name, "film", "actor");
391       Map<String, Object> variables = ImmutableMap.of("title",
392           "Bacon: Shortest Path", "message", "", "actor", name);
393       return new ModelAndView(variables, "bacon_actor.ftl");
394     }
