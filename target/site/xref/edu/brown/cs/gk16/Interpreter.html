16   * This class represents an interpreter that generates a REPL to read in user
17   * input. In its function eval, it then parses the commands given by the user

28    /**
29     * The interpreter has a star object which allows for functionality in
30     * interpreting the commands stars, neighbors, and radius. To add further
31     * functionality, one could create a new object that defines methods to
32     * interpret new commands â€” and each new command could be added to the if else
33     * loop in the eval function which is relatively extensible.
34     */
35    public Interpreter() {
36      stars = new StarsObject(3);
37      ac = new AcObject();
38      bacon = new BaconObject();
39      starsCalled = false;
40      InputStreamReader inputStreamReader = new InputStreamReader(System.in);
41      BufferedReader in = new BufferedReader(inputStreamReader);
42      try {
43        String str;
44        while ((str = in.readLine()) != null) {
45          eval(str);
46        }
47        in.close();
48      } catch (Exception e) {
49        return;
50      }
51    }

53    private int eval(String str) {
54      List<String> commands = tokenizeString(str);
55      int len = commands.size();
56      if (commands.get(0).equals("stars")) {
57        if (len != 2) {
58          return printIncorrectNumberOfArgumentsError();
59        } else {
60          // stars command
61          return doStarsCommand(commands.get(1));
62        }
63      } else if (commands.get(0).equals("neighbors")) {
64        if (parseNeighborsCommand(starsCalled, commands, stars).isEmpty()) {
65          return -1;
66        } else {
67          return 0;
68        }
69      } else if (commands.get(0).equals("radius")) {
70        if (parseRadiusCommand(starsCalled, commands, stars).isEmpty()) {
71          return -1;
72        } else {
73          return 0;
74        }
75      } else if (commands.get(0).equals("corpus")) {
76        if (len != 2) {
77          return printIncorrectNumberOfArgumentsError();
78        } else {
79          return doCorpusCommand(commands.get(1));
80        }
81      } else if (commands.get(0).equals("prefix")
82          || commands.get(0).equals("whitespace")
83          || commands.get(0).equals("smart")) {
84        if (parseToggleCommand(commands, ac).equals("")) {
85          return -1;
86        } else {
87          return 0;
88        }
89      } else if (commands.get(0).equals("led") || commands.get(0).equals("n")) {
90        return parseLedAndNCommand(commands, ac);
91      } else if (commands.get(0).equals("ac")) {
92        doAcCommand(str, ac);
93        return 0;
94      } else if (commands.get(0).equals("mdb")) {
95        if (len != 2) {
96          return printIncorrectNumberOfArgumentsError();
97        } else {
98          return doDatabaseCommand(commands.get(1));
99        }
100     } else if (commands.get(0).equals("connect")) {
101       if (parseConnectCommand(commands, bacon).isEmpty()) {
102         return -1;
103       }
104       return 0;
105     } else {
106       return printCommandNotRecognizedError();
107     }
108   }

110   /**
111    * Tokenizes the passed in string, by splitting on spaces (except for spaces
112    * within quotes).
113    *
114    * @param str
115    *          string to be tokenized
116    * @return list of tokens as strings
117    */
118   public static List<String> tokenizeString(String str) {
119     List<String> tokens = new ArrayList<String>();
120     Matcher m = Pattern.compile("([^\"]\\S*|\".+?\")\\s*").matcher(str);
121     while (m.find()) {
122       tokens.add(m.group(1));
123     }
124     return tokens;
125   }

127   /**
128    * Parses (and error-checks) the arguments passed into the neighbors command
129    * and executes the k-nearest search.
130    *
131    * @param starsLoaded
132    *          boolean detailing whether a stars file has been loaded yet
133    * @param commands
134    *          arraylist containing the neighbors command and its arguments
135    * @param starsOb
136    *          the starsObject to be manipulated
137    * @return list of the k nearest neighbors as string iDs
138    */
139   public static List<String> parseNeighborsCommand(boolean starsLoaded,
140       List<String> commands, StarsObject starsOb) {
141     int length = commands.size();
142     List<String> empty = new ArrayList<String>();
143     if (!starsLoaded) {
144       printNoStarsError();
145       return empty;
146     } else if (length == 3 || length == 5) {
147       int numNeighbors = checkInteger(commands.get(1));
148       if (numNeighbors < 0) {
149         printNotAPositiveIntegerError();
150         return empty;
151       }
152       if (length == 3) {
153         if (!checkStringHasQuotes(commands.get(2))) {
154           printNoQuotesError();
155           return empty;
156         } else {
157           return doNeighborsCommand(numNeighbors,
158               commands.get(2).substring(1, commands.get(2).length() - 1),
159               starsOb);
160         }
161       } else {
162         if (!StarsObject.checkDouble(commands.get(2))
163             || !StarsObject.checkDouble(commands.get(3))
164             || !StarsObject.checkDouble(commands.get(4))) {
165           return empty;
166         } else {
167           double x = StarsObject.makeDouble(commands.get(2));
168           double y = StarsObject.makeDouble(commands.get(3));
169           double z = StarsObject.makeDouble(commands.get(4));
170           return doNeighborsCommand(numNeighbors, x, y, z, starsOb);
171         }
172       }
173     } else {
174       printIncorrectNumberOfArgumentsError();
175       return empty;
176     }
177   }

194     List<String> empty = new ArrayList<String>();
195     if (!starsLoaded) {
196       printNoStarsError();

198     } else if (length == 3 || length == 5) {
199       double radius;
200       if (!StarsObject.checkDouble(commands.get(1))
201           || StarsObject.makeDouble(commands.get(1)) < 0) {
202         printRadiusNotPositiveNumberError();
203         return empty;
204       } else {
205         radius = StarsObject.makeDouble(commands.get(1));
206       }
207       if (length == 3) {
208         if (!checkStringHasQuotes(commands.get(2))) {
209           printNoQuotesError();
210           return empty;
211         } else {
212           return doRadiusCommand(radius,
213               commands.get(2).substring(1, commands.get(2).length() - 1),
214               starsOb);
215         }
216       } else {
217         if (!StarsObject.checkDouble(commands.get(2))
218             || !StarsObject.checkDouble(commands.get(3))
219             || !StarsObject.checkDouble(commands.get(4))) {
220           return empty;
221         } else {
222           double x = StarsObject.makeDouble(commands.get(2));
223           double y = StarsObject.makeDouble(commands.get(3));
224           double z = StarsObject.makeDouble(commands.get(4));
225           return doRadiusCommand(radius, x, y, z, starsOb);

227       }
228     } else {
229       printIncorrectNumberOfArgumentsError();
230       return empty;

234   /**
235    * Parses (and error-checks) the arguments passed into the commands that
236    * toggle certain setting for autocorrect. All such commands should be
237    * followed by no arguments or by "on"/"off".
238    *
239    * @param commands
240    *          arraylist containing the toggle command (and its argument)
241    * @param ac
242    *          the AcObject to be manipulated
243    * @return String detailing toggle setting (or error occurred)
244    */
245   public static String parseToggleCommand(List<String> commands, AcObject ac) {
246     if (commands.size() == 2) {
247       int result = checkToggleArgument(commands.get(1));
248       if (result < 0) {
249         return "";
250       } else {
251         if (commands.get(0).equals("prefix")) {
252           if (result == 1) {
253             ac.setPrefix(true);
254           } else {
255             ac.setPrefix(false);
256           }
257         } else if (commands.get(0).equals("whitespace")) {
258           if (result == 1) {
259             ac.setWhiteSpace(true);
260           } else {
261             ac.setWhiteSpace(false);
262           }
263         } else if (commands.get(0).equals("smart")) {
264           if (result == 1) {
265             ac.setSmart(true);
266           } else {
267             ac.setSmart(false);
268           }
269         } else {
270           printCommandNotRecognizedError();
271           return "";
272         }
273       }
274       return "done";
275     } else if (commands.size() == 1) {
276       if (commands.get(0).equals("prefix")) {
277         System.out.println(ac.getPrefix());
278         return ac.getPrefix();
279       } else if (commands.get(0).equals("whitespace")) {
280         System.out.println(ac.getWhiteSpace());
281         return ac.getWhiteSpace();
282       } else if (commands.get(0).equals("smart")) {
283         System.out.println(ac.getSmart());
284         return ac.getSmart();
285       } else {
286         printCommandNotRecognizedError();
287         return "";
288       }
289     } else {
290       printIncorrectNumberOfArgumentsError();
291       return "";
292     }
293   }

296    * Parses (and error-checks) the led command that sets the levenshtein edit
297    * distance for the autocorrect object or the n command that sets the size of
298    * the n gram to be stored.
299    *
300    * @param commands
301    *          arraylist containing the led command (and its argument)
302    * @param ac
303    *          the AcObject to be manipulated
304    * @return int detailing led or n setting (or error occurred)
305    */
306   public static int parseLedAndNCommand(List<String> commands, AcObject ac) {
307     if (commands.size() == 2) {
308       int val = checkInteger(commands.get(1));
309       if (val < 0) {
310         return printNotAPositiveIntegerError();
311       } else {
312         if (commands.get(0).equals("led")) {
313           ac.setLed(val);
314         } else {
315           ac.setN(val);
316         }
317         return 0;
318       }
319     } else if (commands.size() == 1) {
320       if (commands.get(0).equals("led")) {
321         System.out.println("led " + ac.getLed());
322         return ac.getLed();
323       } else {
324         System.out.println("n " + ac.getN());
325         return ac.getN();
326       }
327     } else {
328       return printIncorrectNumberOfArgumentsError();
329     }
330   }

333    * Parses (and error-checks) the connect command that finds the shortest path
334    * between two actors given a database file.
335    *
336    * @param commands
337    *          arraylist containing the led command (and its argument)

339    *          the BaconObject to be manipulated
340    * @return list of strings detailing the shortest path

342   public static List<String> parseConnectCommand(List<String> commands,
343       BaconObject baconOb) {
344     List<String> empty = new ArrayList<String>();
345     if (commands.size() != 3) {
346       printIncorrectNumberOfArgumentsError();
347       return empty;
348     } else {
349       if (checkStringHasQuotes(commands.get(1))
350           && checkStringHasQuotes(commands.get(2))) {
351         return doConnectCommand(
352             commands.get(1).substring(1, commands.get(1).length() - 1),
353             commands.get(2).substring(1, commands.get(2).length() - 1),
354             baconOb);
355       } else {
356         printNoQuotesError();
357         return empty;
358       }
359     }
360   }

399     if (!string.startsWith("\"")) {
400       return false;

402       return false;
