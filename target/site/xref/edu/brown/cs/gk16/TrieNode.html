1   package edu.brown.cs.gk16;

3   import java.util.HashMap;
4   
5   /**
6    * The representation of a node on a trie which stores its children in a hashmap
7    * that uses characters as keys. Also stores a string and a boolean that details
8    * whether the string is a completed word.
9    */
10  public class TrieNode {
11  
12    private HashMap<Character, TrieNode> children;
13    private boolean isEnd;
14    private String chars;
15    private TrieNode parent;
16  
17    /**
18     * @param characters
19     *          the string to be stored in the node
20     * @param nodeParent
21     *          the node's parent node
22     */
23    public TrieNode(String characters, TrieNode nodeParent) {
24      children = new HashMap<Character, TrieNode>();
25      isEnd = false;
26      chars = characters;
27      parent = nodeParent;
28    }
29  
30    /**
31     * @param end
32     *          boolean detailing whether the contained string is a completed word
33     */
34    public void setEnd(boolean end) {
35      isEnd = end;
36    }
37  
38    /**
39     * @return returns the boolean detailing whether the contained string is a
40     *         completed word
41     */
42    public boolean isEnd() {
43      return isEnd;
44    }
45  
46    /**
47     * @param c
48     *          character that serves as the key to the child node
49     * @param node
50     *          child node to be added to the children hashmap
51     */
52    public void addChild(char c, TrieNode node) {
53      children.put(c, node);
54      return;
55    }
56  
57    /**
58     * @return TrieNode representing the node's parent
59     */
60    public TrieNode getParent() {
61      return parent;
62    }
63  
64    /**
65     * @return hashmap containing the node's children
66     */
67    public HashMap<Character, TrieNode> getChildren() {
68      return children;
69    }
70  
71    /**
72     * @return the string contained in the node
73     */
74    public String getChars() {
75      return chars;
76    }
77  
78    /**
79     * @return the last character of the contained string
80     */
81    public char getChar() {
82      return chars.charAt(chars.length() - 1);
83    }
84  
85  }
